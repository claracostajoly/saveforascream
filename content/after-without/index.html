function startTheVoid() { const now = audioCtx.currentTime; // 1. SOURCE : EROSION NOISE const bufferSize = audioCtx.sampleRate * 4; const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate); const data = buffer.getChannelData(0); // On réduit l'amplitude du bruit initial pour avoir plus de contrôle for (let i = 0; i < bufferSize; i++) { data[i] = (Math.random() * 2 - 1) * 0.5; } const source = audioCtx.createBufferSource(); source.buffer = buffer; source.loop = true; const erosionFilter = audioCtx.createBiquadFilter(); erosionFilter.type = "lowpass"; erosionFilter.frequency.value = 350; // Plus sombre pour moins de fatigue auditive // 2. RÉSEAU DE RÉVERBÉRATION const verbGains = [0.031, 0.057, 0.083, 0.113].map(t => { const d = audioCtx.createDelay(); d.delayTime.value = t; const f = audioCtx.createGain(); f.gain.value = 0.85; // Baisse stratégique pour éviter l'emballement exponentiel return {d, f}; }); // 3. COMPRESSEUR DE MASSE const compressor = audioCtx.createDynamicsCompressor(); compressor.threshold.setValueAtTime(-40, now); compressor.ratio.setValueAtTime(12, now); compressor.attack.setValueAtTime(0.01, now); // Un peu de "mou" pour éviter le clic sec // 4. PROTECTION FINALE (Le Garde-fou) const limiter = audioCtx.createDynamicsCompressor(); limiter.threshold.setValueAtTime(-3.0, now); // On bloque tout à -3dB limiter.ratio.setValueAtTime(20, now); const makeupGain = audioCtx.createGain(); makeupGain.gain.setValueAtTime(0, now); // On remplace le 31.6 par 0.8. // L'effet de "mur de son" viendra de la compression, pas du volume pur. makeupGain.gain.linearRampToValueAtTime(0.8, now + 6); // --- ROUTAGE --- source.connect(erosionFilter); verbGains.forEach(node => { erosionFilter.connect(node.d); node.d.connect(node.f); node.f.connect(node.d); node.d.connect(compressor); }); compressor.connect(makeupGain); makeupGain.connect(limiter); limiter.connect(audioCtx.destination); source.start(); }
