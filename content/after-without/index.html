function startAfterWithout() { ctx.resume(); // Source : Bruit de fond très fin const whiteNoise = ctx.createBufferSource(); const bufferSize = 2 * ctx.sampleRate; const noiseBuffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate); const data = noiseBuffer.getChannelData(0); for (let i = 0; i < bufferSize; i++) { data[i] = Math.random() * 0.005; } whiteNoise.buffer = noiseBuffer; whiteNoise.loop = true; // Simulation CORPUS (Comb Filter) // Crée la résonance métallique "Beam" const delay = ctx.createDelay(); delay.delayTime.value = 0.0022; // Fréquence de la résonance const feedback = ctx.createGain(); feedback.gain.value = 0.95; // Intensité de la résonance // REVERB 100% WET (Convolver ou Delay long) const reverb = ctx.createConvolver(); // Pour un son massif // (Alternative simple : un feedback delay long) // COMPRESSION + GAIN (+30dB comme sur ta capture) const compressor = ctx.createDynamicsCompressor(); compressor.threshold.setValueAtTime(-50, ctx.currentTime); compressor.ratio.setValueAtTime(12, ctx.currentTime); const makeUpGain = ctx.createGain(); makeUpGain.gain.value = 40; // Routage en boucle pour le Corpus whiteNoise.connect(delay); delay.connect(feedback); feedback.connect(delay); // Sortie vers la chaîne finale delay.connect(compressor); compressor.connect(makeUpGain); makeUpGain.connect(ctx.destination); whiteNoise.start(); }
